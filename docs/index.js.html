'use strict';

const Brick = require('cta-brick');
const defaultLogger = require('cta-logger');

class MyClass extends Brick {

  /**
   * constructor - MyClass constructor
   *
   * @param  {type} cementHelper this is an cementHelper
   * @param  {type} config       this is an config
   */
  constructor(cementHelper, config) {
    super(cementHelper, config);

    this.logger = defaultLogger();
  }
  validate(context) {
    this.logger.info('context:', context);
  }
  process(context) {
    this.logger.info('context:', context);
  }
}

module.exports = MyClass;
